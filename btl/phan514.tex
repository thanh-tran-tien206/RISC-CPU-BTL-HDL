\subsubsection{Testbench và Kết quả mô phỏng Khối điều khiển (Controller)}

\textbf{a. Mô tả kịch bản kiểm chứng (Testbench)} \\
Môi trường mô phỏng khối Controller được thiết lập để kiểm tra chu trình máy trạng thái (FSM) và sự chính xác của các tín hiệu điều khiển đầu ra. Các kịch bản bao gồm:
\begin{itemize}
    \item \textbf{Khởi tạo và Fetch}: Kiểm tra quá trình reset hệ thống  và 4 trạng thái đầu (\texttt{INST\_ADDR} đến \texttt{IDLE}) xem các tín hiệu \texttt{sel}, \texttt{rd}, \texttt{ld\_ir} có kích hoạt đúng để nạp lệnh từ bộ nhớ hay không.
    \item \textbf{Lệnh dừng (HALT)}: Kiểm tra khi \texttt{opcode = 3'b000}, tín hiệu \texttt{halt} phải được bật tại trạng thái \texttt{OP\_ADDR}.
    \item \textbf{Lệnh tính toán (ADD/LDA/AND/SUB)}: Với các mã lệnh từ \texttt{3'b010} đến \texttt{3'b101}, kiểm tra việc bật tín hiệu \texttt{rd} ở giai đoạn thực thi và tín hiệu \texttt{ld\_ac} ở trạng thái \texttt{STORE}.
    \item \textbf{Lệnh rẽ nhánh (SKZ)}: Với \texttt{opcode = 3'b001}, kiểm tra tín hiệu \texttt{inc\_pc} có bật thêm ở trạng thái \texttt{ALU\_OP} khi cờ \texttt{zero = 1} hay không.
    \item \textbf{Lệnh lưu trữ (STO)}: Với \texttt{opcode = 3'b110}, kiểm tra sự phối hợp giữa \texttt{data\_e} và \texttt{wr} để ghi dữ liệu ra bộ nhớ tại trạng thái \texttt{STORE}.
\end{itemize}

\textbf{b. Mã nguồn Testbench (Verilog)}
\begin{lstlisting}[language=Verilog, caption={Mã nguồn Testbench kiểm tra khối Controller}]
`timescale 1ns / 1ps

module controller_tb;
    // Tin hieu dau vao (Inputs)
    reg clk;
    reg rst;
    reg [2:0] opcode;
    reg is_zero;

    // Tin hieu dau ra (Outputs)
    wire sel, rd, ld_ir, halt, inc_pc, ld_ac, ld_pc, wr, data_e;
    wire [2:0] debug_state;

    // Khoi tao don vi can kiem tra (UUT)
    controller uut (
        .clk(clk), .rst(rst), .opcode(opcode), .is_zero(is_zero), 
        .sel(sel), .rd(rd), .ld_ir(ld_ir), .halt(halt), 
        .inc_pc(inc_pc), .ld_ac(ld_ac), .ld_pc(ld_pc), 
        .wr(wr), .data_e(data_e), .debug_state(debug_state)
    );

    // Tao xung Clock chu ky 10ns
    always #5 clk = ~clk;

    initial begin
        // Khoi tao ban dau
        clk = 0; rst = 1; opcode = 3'b000; is_zero = 0;
        
        // Giai doan Reset
        #15 rst = 0;

        // --- Kich ban 1: Kiem tra chu ky Fetch va lenh HALT ---
        @(posedge clk); // INST_ADDR 
        @(posedge clk); // INST_FETCH 
        @(posedge clk); // INST_LOAD 
        @(posedge clk); // IDLE 
        opcode = 3'b000; 
        @(posedge clk); // OP_ADDR -> halt se len 1 
        
        // --- Kich ban 2: Lenh ADD/LOAD (opcode 010) ---
        rst = 1; #10 rst = 0; 
        opcode = 3'b010; 
        wait(debug_state == 3'd6); // ALU_OP: rd = 1 
        #10;
        wait(debug_state == 3'd7); // STORE: ld_ac = 1 
        #10;

        // --- Kich ban 3: Lenh STORE (opcode 110) ---
        opcode = 3'b110;
        wait(debug_state == 3'd7); // STORE: wr = 1, data_e = 1

        #50;
        $stop; 
    end
      
    initial begin
        $monitor("Time=%0t | State=%d | Opcode=%b | Halt=%b | Rd=%b | Wr=%b | Ld_AC=%b", 
                 $time, debug_state, opcode, halt, rd, wr, ld_ac);
    end
endmodule
\end{lstlisting}

\textbf{Đánh giá kết quả:}
\begin{itemize}
    \item \textbf{Tính tuần tự}: Tín hiệu \texttt{debug\_state} thay đổi liên tục theo đúng chu trình thiết kế từ 0 đến 7.
    \item \textbf{Đúng đắn về logic}: Tín hiệu \texttt{halt} kích hoạt chính xác khi gặp opcode dừng. Tín hiệu \texttt{rd} bật đúng lúc ở các giai đoạn nạp và thực thi lệnh.
    \item \textbf{Cơ chế nạp/ghi}: Tín hiệu \texttt{ld\_ac} và \texttt{wr} chỉ xuất hiện ở bước cuối (\texttt{STORE}), đảm bảo đồng bộ dữ liệu hệ thống trước khi lưu trữ.
\end{itemize}
